python-tkinter 具有“复制、粘贴、删除”等功能的菜单

发表于 2020-12-19 更新于 2021-01-05 分类于 Python
  • .event_generate(self,sequence,kw) 有人称为“被忽略的命令”
  • 代码中尖括号里的参数还有没有其它的？没有查到资料，遗憾。

代码1如下：

from tkinter import *

def cut(event=None):
    texteditor.event_generate("<<Cut>>")  # 双尖括号
def copy(event=None):
    texteditor.event_generate("<<Copy>>")  # 双尖括号
def paste(event=None):
    texteditor.event_generate('<<Paste>>'  # 双尖括号
def delete(event=None):
    texteditor.event_generate('<Delete>')  # 单尖括号
def backspace(event=None):
    texteditor.event_generate('<BackSpace>')  # 单尖括号

root=Tk()
texteditor=Text(root)
texteditor.pack(fill=BOTH)
menubar=Menu(root)
filemenu=Menu(menubar)
root.config(menu=menubar)

filemenu.add_command(label="剪切",command=cut)
filemenu.add_command(label="复制",command=copy)
filemenu.add_command(label="粘贴",command=paste)
filemenu.add_command(label="删除",command=delete)
filemenu.add_command(label="退格",command=backspace)

menubar.add_cascade(label="File",menu=filemenu)

root.mainloop()

代码2如下：

import tkinter as tk

window = tk.Tk()
window.title('右键菜单')
window.geometry('400x260')

def text_edit_cut():
    text_edit.event_generate("<<Cut>>")  # 双尖括号
def text_edit_copy():
    text_edit.event_generate("<<Copy>>")  # 双尖括号
def text_edit_paste():
    text_edit.event_generate('<<Paste>>')  # 双尖括号
def text_edit_delete():
    text_edit.event_generate('<Delete>')  # 单尖括号
def text_edit_backspace():
    text_edit.event_generate('<BackSpace>')  # 单尖括号

def text_edit_right(event) :  # 调出右键菜单
    popupbar_text.post(event.x_root, event.y_root)  # 激活右键菜单

popupbar_text = tk.Menu(window, tearoff=False)
popupbar_text.add_command(label="剪切",command=text_edit_cut)
popupbar_text.add_command(label="复制",command=text_edit_copy)
popupbar_text.add_command(label="粘贴",command=text_edit_paste)
popupbar_text.add_separator()    # 分隔符
popupbar_text.add_command(label="删除",command=text_edit_delete)
popupbar_text.add_separator()    # 分隔符
popupbar_text.add_command(label="退格",command=text_edit_backspace)

text_edit=tk.Text(window)  # 创建多行文本框用以显示右键菜单
text_edit.pack()

####为文本框text_edit设置绑定事件
text_edit.bind("<ButtonRelease-3>",text_edit_right)  # 鼠标右键点击触发事件

window.mainloop()
